Publishers.fromJson(dynamic json) {
  id = json['id'];
  display = json['display'];
  audioCodec = json['audio_codec'];
  videoCodec = json['video_codec'];
  if (json['streams'] != null) {
    streams = [];
    json['streams'].forEach((v) {
      streams?.add(Streams.fromMap(v));
    });
  }
  talking = json['talking'];
}
