StreamingMountPoint.fromJson(dynamic json) {
  id = json['id'];
  type = json['type'];
  description = json['description'];
  metadata = json['metadata'];
  enabled = json['enabled'];
  if (json['media'] != null) {
    media = [];
    json['media'].forEach((v) {
      media?.add(Media.fromJson(v));
    });
  }
}
